        function toggleMenu() {

            document.getElementById('navContainer').classList.toggle('mobile-active');

        }


        // --- HLAVNÍ FUNKCE PRO PŘEPÍNÁNÍ ---

        function tab(id, event) {

            if (event) event.preventDefault();

            

            const urlPath = Object.keys(routeMap).find(key => routeMap[key] === id && key !== '');

            const newPath = urlPath ? '/' + urlPath : '/';

            

            window.history.pushState({sectionId: id}, '', newPath);

            showSection(id);

        }


        function showSection(id) {

            document.querySelectorAll('.section').forEach(s => s.classList.remove('active'));

            document.querySelectorAll('nav a').forEach(a => a.classList.remove('active'));

            

            const targetSection = document.getElementById(id);

            const targetNav = document.getElementById('t-' + id);

            

            if (targetSection) targetSection.classList.add('active');

            if (targetNav) targetNav.classList.add('active');

            

            document.getElementById('navContainer').classList.remove('mobile-active');

            window.scrollTo({top: 0, behavior: 'smooth'});

        }


        window.addEventListener('popstate', (event) => {

            if (event.state && event.state.sectionId) {

                showSection(event.state.sectionId);

            } else {

                handleInitialRouting();

            }

        });


        function handleInitialRouting() {

            const path = window.location.pathname.replace(/^\//, '');

            const sectionId = routeMap[path] || 'home';

            showSection(sectionId);

        }

        function openLightbox(idx) {

        }

        function closeLightbox() {

            document.getElementById('lightbox').style.display = 'none';

            document.body.style.overflow = 'auto';

        }
